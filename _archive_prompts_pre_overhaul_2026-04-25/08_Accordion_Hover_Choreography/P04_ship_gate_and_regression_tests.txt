§C — PLAYWRIGHT RAPID-SWAP SCENARIOS

C1. Cold 5-tab traversal (no expansions)

    Goal: a cursor passing over all panels in <600ms
    total produces ZERO expansions.

    Script sketch (tests/accordion/rapid_swap.spec.ts):

      test("C1: cold 5-tab traversal produces no expansion", async ({ page }) => {
        await page.goto("/");
        await expect(page.locator(".accordion")).toBeVisible();
        // Ensure the accordion is in neutral mode. If
        // initialExpandedIndex=1 is the default, press
        // ESC first.
        await page.keyboard.press("Escape");

        // Read panel bounding boxes.
        const spines = page.locator(".panel-spine");
        const count = await spines.count();
        expect(count).toBeGreaterThanOrEqual(5);

        // Record activeIndex at start.
        const startState = await page.evaluate(() =>
          document.querySelector(".accordion")
            ?.querySelector<HTMLElement>('[data-state="expanded"]')
            ?.id ?? null
        );
        expect(startState).toBeNull();

        // Traverse cursor across all 5 spines in 300ms
        // total. Use page.mouse.move with steps to
        // interpolate smoothly.
        const boxes = await Promise.all(
          Array.from({ length: 5 }, (_, i) => spines.nth(i).boundingBox())
        );
        await page.mouse.move(boxes[0]!.x + 10, boxes[0]!.y + 10);
        for (let i = 1; i < 5; i++) {
          await page.mouse.move(
            boxes[i]!.x + 10,
            boxes[i]!.y + 10,
            { steps: 8 }
          );
          await page.waitForTimeout(40);  // 40ms per hop;
                                          // total 160ms
                                          // for 4 hops
        }
        // Total elapsed: ~160ms + move durations ≈ 300ms.

        // Assert no panel became expanded.
        const endState = await page.evaluate(() =>
          document.querySelector(".accordion")
            ?.querySelector<HTMLElement>('[data-state="expanded"]')
            ?.id ?? null
        );
        expect(endState).toBeNull();
      });

    Pass criteria: the test must pass deterministically
    on Chromium, WebKit, and Firefox. If Firefox's
    pointer-event timing differs enough to cause
    sub-120ms dwells to occasionally register, increase
    the per-hop wait to 50ms (total 500ms; still well
    under a real-user's accidental dwell).

C3. Warm swap: no neutral gap

    test("C3: warm swap N→M has no neutral intermediate", async ({ page }) => {
      await page.goto("/");
      // Dwell on panel 2.
      const spine2 = page.locator(".panel-spine").nth(1);
      const spine4 = page.locator(".panel-spine").nth(3);
      const box2 = await spine2.boundingBox();
      const box4 = await spine4.boundingBox();
      await page.mouse.move(box2!.x + 10, box2!.y + 10);
      await page.waitForTimeout(200);  // now warm on 2.

      // Observe data-state flips during swap. Use CDP to
      // capture rendered frames, OR poll the DOM at 16ms
      // intervals for 600ms.
      const states: (number | null)[] = [];
      const polling = page.evaluate(() => {
        return new Promise<(number | null)[]>((resolve) => {
          const observed: (number | null)[] = [];
          const interval = setInterval(() => {
            const expanded = document.querySelector(
              '.accordion [data-state="expanded"]'
            ) as HTMLElement | null;
            if (!expanded) { observed.push(null); return; }
            const m = /panel-body-0?(\d)/.exec(expanded.id);
            observed.push(m ? parseInt(m[1], 10) : null);
          }, 16);
          setTimeout(() => {
            clearInterval(interval);
            resolve(observed);
          }, 600);
        });
      });

      // Trigger swap.
      await page.mouse.move(box4!.x + 10, box4!.y + 10);
      const observed = await polling;

      // Assert: the sequence never contains a null
      // between the last 2 and the first 4. (Invariant
      // I5 from P01.)
      const last2 = observed.lastIndexOf(2);
      const first4 = observed.indexOf(4);
      expect(first4).toBeGreaterThan(last2);
      for (let i = last2 + 1; i < first4; i++) {
        // Between the last observed "2" and the first
        // observed "4", all frames must show either 2 or
        // 4 (never null).
        expect([2, 4]).toContain(observed[i]);
      }
    });

C4. Mid-animation swap: no opacity restart

    Goal: panel N mid-fade-in swapped to panel M must
    NOT show panel N jumping to opacity 1 at any frame.

    test("C4: mid-animation swap produces monotonic opacity", async ({ page }) => {
      await page.goto("/");
      await page.keyboard.press("Escape");

      const spine2 = page.locator(".panel-spine").nth(1);
      const spine3 = page.locator(".panel-spine").nth(2);
      const box2 = await spine2.boundingBox();
      const box3 = await spine3.boundingBox();

      // Start cold. Hover on panel 2.
      await page.mouse.move(box2!.x + 10, box2!.y + 10);

      // Dwell elapses at 120ms; fade-in-delay ends at
      // 120+260=380ms; fade-in completes at 380+260=640ms.
      // Swap at t=500 — panel 2's body should be at
      // opacity ~0.46 (linear from 0 at t=380 to 1 at
      // t=640; at t=500 → (500-380)/260 ≈ 0.46, or
      // easier with ease-out-quint).

      // Sample opacity of panel-2 body at 20ms intervals
      // through the swap window.
      const samples: Array<{ t: number; opacity: number }> = [];
      const sampler = page.evaluate(async () => {
        const body = document.querySelector(
          '.accordion #panel-body-02 .panel-body'
        ) as HTMLElement | null;
        if (!body) return [];
        const out: Array<{ t: number; opacity: number }> = [];
        const start = performance.now();
        return new Promise<Array<{ t: number; opacity: number }>>((resolve) => {
          const tick = () => {
            const cs = getComputedStyle(body);
            out.push({
              t: performance.now() - start,
              opacity: parseFloat(cs.opacity),
            });
            if (performance.now() - start < 800) {
              requestAnimationFrame(tick);
            } else {
              resolve(out);
            }
          };
          tick();
        });
      });

      await page.waitForTimeout(500);
      await page.mouse.move(box3!.x + 10, box3!.y + 10);
      const trace = await sampler;

      // Find the swap moment (opacity starts decreasing
      // after reaching its peak). Verify that from the
      // peak onward, opacity is monotonically
      // non-increasing (no spike back up to 1).
      let peak = 0;
      let peakIdx = 0;
      for (let i = 0; i < trace.length; i++) {
        if (trace[i].opacity > peak) {
          peak = trace[i].opacity;
          peakIdx = i;
        }
      }
      for (let i = peakIdx + 1; i < trace.length; i++) {
        // Tolerate 0.02 noise for compositor rounding.
        expect(trace[i].opacity).toBeLessThanOrEqual(
          trace[i - 1].opacity + 0.02
        );
      }

      // Also: peak opacity should be < 0.7 (it was
      // mid-fade-in, interrupted before reaching 1).
      // If peak is 1.0, that means the body fully faded
      // in before the swap — test timing is off; reduce
      // waitForTimeout.
      expect(peak).toBeLessThan(0.7);
      expect(peak).toBeGreaterThan(0.1);  // sanity: the
                                          // fade-in did
                                          // happen
    });

    If jsdom's computed-style machinery for transitions
    differs from real Chromium, this test must run in a
    headed Playwright browser, not in jsdom. Configure
    accordingly.

                                     // allow 17ms

    Frame budget assertions are hardware-sensitive. Set

    retains 17ms.

    Expose a `--strict` flag in ship_gate.ts that
    enforces 17ms; without the flag, 20ms is tolerated.

      const files = fs.readdirSync(chunkDir)
        .filter(f => f.endsWith(".js"))
        .map(f => path.join(chunkDir, f));

      let totalGz = 0;
      for (const f of files) {
        const content = fs.readFileSync(f);
        if (content.toString("utf8").includes("useHoverIntent")) {
          totalGz += zlib.gzipSync(content).length;
        }
      }

      // 3. Assert against budget.
      const BUDGET_BYTES = 3200;  // 3.2KB (P01 2.6 +
                                  // P02 0.3 + P03 0.5
                                  // - shared overhead).
      if (totalGz > BUDGET_BYTES) {
        console.error(`Bundle budget exceeded: ${totalGz}B > ${BUDGET_BYTES}B`);
        process.exit(1);
      }
      console.log(`Bundle budget OK: ${totalGz}B / ${BUDGET_BYTES}B`);

    This is approximate — the gzipped size of a chunk
    that CONTAINS useHoverIntent is larger than the
    incremental contribution of hoverIntent alone. A
    better method diffs against a baseline recorded
    pre-08: run `git stash && pnpm build && record
    baseline && git stash pop && pnpm build && compare`.
    Ship_gate.ts can orchestrate this.

    For this prompt's acceptance, the approximate check
    is sufficient to catch 10x overruns. If a finer
    diff is wanted, follow up with a dedicated bundle-
    analyzer integration.

      import { spawnSync } from "node:child_process";

      const steps: Array<{ name: string; cmd: string[] }> = [
        { name: "Unit: hover intent", cmd: ["pnpm", "vitest", "run", "tests/accordion/hover_intent.test.tsx"] },
        { name: "Unit: R01/R02 invariants", cmd: ["pnpm", "vitest", "run", "tests/accordion/r01_r02_invariants.test.ts"] },
        { name: "E2E: rapid swap", cmd: ["pnpm", "playwright", "test", "tests/accordion/rapid_swap.spec.ts"] },
        { name: "E2E: a11y", cmd: ["pnpm", "playwright", "test", "tests/accordion/a11y.spec.ts"] },
        { name: "Lint: will-change hygiene", cmd: ["node", "tests/accordion/will_change_hygiene.ts"] },
        { name: "Build + bundle budget", cmd: ["node", "tests/accordion/bundle_budget.ts"] },
      ];

      const results: Array<{ name: string; ok: boolean; ms: number }> = [];
      for (const step of steps) {
        const t0 = Date.now();
        const r = spawnSync(step.cmd[0], step.cmd.slice(1), {
          stdio: "inherit",
        });
        const ok = r.status === 0;
        results.push({ name: step.name, ok, ms: Date.now() - t0 });
        if (!ok) break;  // fail fast
      }

      console.log("\n=== 08 Ship Gate Summary ===");
      for (const r of results) {
        console.log(`  ${r.ok ? "✓" : "✗"} ${r.name} (${r.ms}ms)`);
      }
      const allOk = results.every(r => r.ok) && results.length === steps.length;
      process.exit(allOk ? 0 : 1);
